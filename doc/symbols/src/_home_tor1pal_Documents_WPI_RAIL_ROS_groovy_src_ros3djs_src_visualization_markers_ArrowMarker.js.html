1 ROS3D.ArrowMarker = function(options) {
  2   var origin = options.origin || new THREE.Vector3(0, 0, 0);
  3   var direction = options.direction || new THREE.Vector3(1, 0, 0);
  4 
  5   var length = options.length || 1;
  6   var headLength = options.headLength || 0.2;
  7 
  8   var shaftDiameter = options.shaftDiameter || 0.05;
  9   var headDiameter = options.headDiameter || 0.1;
 10 
 11   var material = options.material || new THREE.MeshBasicMaterial();
 12 
 13   var shaftLength = length - headLength;
 14 
 15   // create and merge geometry
 16   var geometry = new THREE.CylinderGeometry(shaftDiameter * 0.5, shaftDiameter * 0.5, shaftLength,
 17       12, 1);
 18 
 19   var m = new THREE.Matrix4;
 20   m.setPosition(new THREE.Vector3(0, shaftLength * 0.5, 0));
 21   geometry.applyMatrix(m);
 22 
 23   var coneGeometry = new THREE.CylinderGeometry(0, headDiameter * 0.5, headLength, 12, 1);
 24 
 25   m.setPosition(new THREE.Vector3(0, shaftLength + (headLength * 0.5), 0));
 26   coneGeometry.applyMatrix(m);
 27 
 28   THREE.GeometryUtils.merge(geometry, coneGeometry);
 29 
 30   THREE.Mesh.call(this, geometry, material);
 31 
 32   this.position = origin;
 33   this.setDirection(direction);
 34 };
 35 ROS3D.ArrowMarker.prototype = Object.create(THREE.Mesh.prototype);
 36 
 37 ROS3D.ArrowMarker.prototype.setDirection = function(direction) {
 38 
 39   var axis = new THREE.Vector3(0, 1, 0).cross(direction);
 40 
 41   var radians = Math.acos(new THREE.Vector3(0, 1, 0).dot(direction.clone().normalize()));
 42 
 43   this.matrix = new THREE.Matrix4().makeRotationAxis(axis.normalize(), radians);
 44 
 45   this.rotation.setEulerFromRotationMatrix(this.matrix, this.eulerOrder);
 46 
 47 };
 48 
 49 ROS3D.ArrowMarker.prototype.setLength = function(length) {
 50   this.scale.set(length, length, length);
 51 };
 52 
 53 ROS3D.ArrowMarker.prototype.setColor = function(hex) {
 54   this.line.material.color.setHex(hex);
 55   this.cone.material.color.setHex(hex);
 56 };
