 31 /**
 32  * Create a THREE material based on the given RGBA values.
 33  * 
 34  * @param r - the red value
 35  * @param g - the green value
 36  * @param b - the blue value
 37  * @param a - the alpha value
 38  * @returns the THREE material
 39  */
 40 ROS3D.makeColorMaterial = function(r, g, b, a) {
 41   var color = new THREE.Color();
 42   color.setRGB(r, g, b);
 43   if (a <= 0.99) {
 44     return new THREE.MeshBasicMaterial({
 45       color : color.getHex(),
 46       opacity : a + 0.1,
 47       transparent : true,
 48       depthWrite : true,
 49       blendSrc : THREE.SrcAlphaFactor,
 50       blendDst : THREE.OneMinusSrcAlphaFactor,
 51       blendEquation : THREE.ReverseSubtractEquation,
 52       blending : THREE.NormalBlending
 53     });
 54   } else {
 55     return new THREE.MeshLambertMaterial({
 56       color : color.getHex(),
 57       opacity : a,
 58       blending : THREE.NormalBlending
 59     });
 60   }
 61 };
