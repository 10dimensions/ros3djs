 42 var findClosestPoint = function(target_ray, mouse_ray) {
 43   // Find the closest point on target_ray to any point on mouse_ray.
 44   // Math taken from http://paulbourke.net/geometry/lineline3d/
 45   // line P1->P2 is target_ray
 46   // line P3->P4 is mouse_ray
 47 
 48   var v13 = new THREE.Vector3;
 49   v13.subVectors(target_ray.origin, mouse_ray.origin);
 50   var v43 = mouse_ray.direction.clone();
 51   var v21 = target_ray.direction.clone();
 52   var d1343 = v13.dot(v43);
 53   var d4321 = v43.dot(v21);
 54   var d1321 = v13.dot(v21);
 55   var d4343 = v43.dot(v43);
 56   var d2121 = v21.dot(v21);
 57 
 58   var denom = d2121 * d4343 - d4321 * d4321;
 59   if (Math.abs(denom) <= 0.0001) {
 60     return undefined;
 61   }
 62   var numer = d1343 * d4321 - d1321 * d4343;
 63 
 64   var mua = numer / denom;
 65   return mua;
 66 };

 68 var closestAxisPoint = function(axisRay, camera, mousePos) {
 69   // project axis onto screen
 70   var o = axisRay.origin.clone();
 71   projector.projectVector(o, camera);
 72 
 73   var o2 = axisRay.direction.clone().add(axisRay.origin);
 74   projector.projectVector(o2, camera);
 75 
 76   // d is the axis vector in screen space
 77   var d = o2.clone().sub(o);
 78   // d = o2-o;
 79 
 80   // t is the 2d ray param of perpendicular projection
 81   // of mousePos onto o
 82   var tmp = new THREE.Vector2;
 83   var t = tmp.subVectors(mousePos, o).dot(d) / d.dot(d);
 84   // t = (mousePos - o) * d / (d*d);
 85 
 86   // mp is the final 2d-projected mouse pos DDD
 87   var mp = new THREE.Vector2;
 88   mp.addVectors(o, d.clone().multiplyScalar(t));
 89   // mp = o + d*t;
 90 
 91   // go back to 3d by shooting a ray
 92   var vector = new THREE.Vector3(mp.x, mp.y, 0.5);
 93   projector.unprojectVector(vector, camera);
 94   var mpRay = new THREE.Ray(camera.position, vector.sub(camera.position).normalize());
 95   var mua = findClosestPoint(axisRay, mpRay, mua);
 96 
 97   return mua;
 98 };

100 var intersectPlane = function(mouseRay, planeOrigin, planeNormal) {
101 
102   var vector = new THREE.Vector3();
103   var intersectPoint = new THREE.Vector3();
104 
105   vector.subVectors(planeOrigin, mouseRay.origin);
106   dot = mouseRay.direction.dot(planeNormal);
107 
108   // bail if ray and plane are parallel
109   if (Math.abs(dot) < mouseRay.precision)
110     return null;
111 
112   // calc distance to plane
113   scalar = planeNormal.dot(vector) / dot;
114 
115   // if negative distance, then plane is behind ray
116   // if (scalar < 0)
117   // return null;
118 
119   intersectPoint.addVectors(mouseRay.origin, mouseRay.direction.clone().multiplyScalar(scalar));
120   return intersectPoint;
121 };

129 InteractiveMarker.prototype.moveAxis = function(control, origAxis, event3d) {
130   if (this.dragging) {
131     var currentControlOri = control.currentControlOri;
132     var axis = origAxis.clone().applyQuaternion(currentControlOri);
133     // get move axis in world coords
134     var originWorld = this.dragStart.event3d.intersection.point;
135     var axisWorld = axis.clone().applyQuaternion(this.dragStart.orientationWorld.clone());
136 
137     var axisRay = new THREE.Ray(originWorld, axisWorld);
138 
139     // find closest point to mouse on axis
140     var t = closestAxisPoint(axisRay, event3d.camera, event3d.mousePos);
141 
142     // offset from drag start position
143     var p = new THREE.Vector3;
144     p.addVectors(this.dragStart.position, axis.clone().applyQuaternion(this.dragStart.orientation)
145         .multiplyScalar(t));
146     this.setPosition(control, p);
147 
148     event3d.stopPropagation();
149   }
150 };

152 InteractiveMarker.prototype.movePlane = function(control, origNormal, event3d) {
153   if (this.dragging) {
154     var currentControlOri = control.currentControlOri;
155     var normal = origNormal.clone().applyQuaternion(currentControlOri);
156     // get plane params in world coords
157     var originWorld = this.dragStart.event3d.intersection.point;
158     var normalWorld = normal.clone.applyQuaternion(this.dragStart.orientationWorld);
159 
160     // intersect mouse ray with plane
161     var intersection = intersectPlane(event3d.mouseRay, originWorld, normalWorld);
162 
163     // offset from drag start position
164     var p = new THREE.Vector3;
165     p.subVectors(intersection, originWorld);
166     p.add(this.dragStart.positionWorld);
167     this.setPosition(control, p);
168     event3d.stopPropagation();
169   }
170 };

179 InteractiveMarker.prototype.rotateAxis = function(control, origOrientation, event3d) {
180   if (this.dragging) {
181     control.updateMatrixWorld();
182 
183     // console.log("------------------_");
184 
185     var currentControlOri = control.currentControlOri;
186     var orientation = currentControlOri.clone().multiply(origOrientation.clone());
187 
188     printQuat(currentControlOri);
189     printQuat(orientation);
190 
191     var normal = (new THREE.Vector3(1, 0, 0)).applyQuaternion(orientation);
192 
193     // get plane params in world coords
194     var originWorld = this.dragStart.event3d.intersection.point;
195     var normalWorld = normal.applyQuaternion(this.dragStart.orientationWorld);
196 
197     printVec(normal);
198     printVec(normalWorld);
199 
200     // intersect mouse ray with plane
201     var intersection = intersectPlane(event3d.mouseRay, originWorld, normalWorld);
202 
203     // offset local origin to lie on intersection plane
204     var normalRay = new THREE.Ray(this.dragStart.positionWorld, normalWorld);
205     var rotOrigin = intersectPlane(normalRay, originWorld, normalWorld);
206 
207     // rotates from world to plane coords
208     var orientationWorld = this.dragStart.orientationWorld.clone().multiply(orientation);
209     var orientationWorldInv = orientationWorld.clone().inverse();
210 
211     // rotate original and current intersection into local coords
212     intersection.sub(rotOrigin);
213     intersection.applyQuaternion(orientationWorldInv);
214 
215     var origIntersection = this.dragStart.event3d.intersection.point.clone();
216     origIntersection.sub(rotOrigin);
217     origIntersection.applyQuaternion(orientationWorldInv);
218 
219     // compute relative 2d angle
220     var a1 = Math.atan2(intersection.y, intersection.z);
221     var a2 = Math.atan2(origIntersection.y, origIntersection.z);
222     var a = a2 - a1;
223 
224     var rot = new THREE.Quaternion();
225     rot.setFromAxisAngle(normal, a);
226 
227     // rotate
228     // this.setOrientation( rot.multiplySelf(this.dragStart.orientationWorld) );
229     this.setOrientation(control, rot.multiply(this.dragStart.orientationWorld));
230 
231     // offset from drag start position
232     event3d.stopPropagation();
233   }
234 };

245 InteractiveMarker.prototype.startDrag = function(control, event3d) {
246   if (event3d.domEvent.button !== 0) {
247     return;
248   }
249   event3d.stopPropagation();
250   this.dragging = true;
251   this.updateMatrixWorld(true);
252   var scale = new THREE.Vector3();
253   this.matrixWorld.decompose(this.dragStart.positionWorld, this.dragStart.orientationWorld, scale);
254   this.dragStart.position = this.position.clone();
255   this.dragStart.orientation = this.quaternion.clone();
256   this.dragStart.event3d = event3d;
257 
258   this.feedbackEvent("user_mouse_down", control);
259 };

261 InteractiveMarker.prototype.stopDrag = function(control, event3d) {
262   if (event3d.domEvent.button !== 0) {
263     return;
264   }
265   event3d.stopPropagation();
266   this.dragging = false;
267   this.dragStart.event3d = {};
268   this.onServerSetPose(this.bufferedPoseEvent);
269   this.bufferedPoseEvent = undefined;
270 
271   this.feedbackEvent("user_mouse_up", control);
272 };

290 InteractiveMarker.prototype.onServerSetPose = function(event) {
291   if (event === undefined) {
292     return;
293   }
294 
295   if (this.dragging) {
296     this.bufferedPoseEvent = event;
297     return;
298   }
299 
300   var pose = event.pose;
301 
302   this.position.x = pose.position.x;
303   this.position.y = pose.position.y;
304   this.position.z = pose.position.z;
305 
306   this.useQuaternion = true;
307   this.quaternion = new THREE.Quaternion(pose.orientation.x, pose.orientation.y,
308       pose.orientation.z, pose.orientation.w);
309 
310   this.updateMatrixWorld(true);
311 };

315 var InteractiveMarkerControl = ImThree.InteractiveMarkerControl = function(parent, controlMsg,
316     camera, meshBaseUrl) {
317   THREE.Object3D.call(this);
318   THREE.EventDispatcher.call(this);
319 
320   this.parent = parent;
321   this.dragging = false;
322 
323   var that = this;
324 
325   this.name = controlMsg.name;
326 
327   var NONE = 0;
328   var MENU = 1;
329   var BUTTON = 2;
330   var MOVE_AXIS = 3;
331   var MOVE_PLANE = 4;
332   var ROTATE_AXIS = 5;
333   var MOVE_ROTATE = 6;
334 
335   var controlOri = new THREE.Quaternion(controlMsg.orientation.x, controlMsg.orientation.y,
336       controlMsg.orientation.z, controlMsg.orientation.w);
337   controlOri.normalize();
338 
339   // transform x axis into local frame
340   var controlAxis = new THREE.Vector3(1, 0, 0);
341   controlAxis.applyQuaternion(controlOri);
342 
343   this.currentControlOri = new THREE.Quaternion();
344 
345   // determine mouse interaction
346   switch (controlMsg.interaction_mode) {
347     case MOVE_AXIS:
348       this.addEventListener("mousemove", parent.moveAxis.bind(parent, this, controlAxis));
349       this.addEventListener("touchmove", parent.moveAxis.bind(parent, this, controlAxis));
350       break;
351     case ROTATE_AXIS:
352       this.addEventListener("mousemove", parent.rotateAxis.bind(parent, this, controlOri));
353       break;
354     case MOVE_PLANE:
355       this.addEventListener("mousemove", parent.movePlane.bind(parent, this, controlAxis));
356       break;
357     case BUTTON:
358       this.addEventListener("click", parent.buttonClick.bind(parent, this));
359       break;
360     default:
361       break;
362   }
363 
364   // install default listeners for highlighting / dragging
365   function stopPropagation(event) {
366     event.stopPropagation();
367   }
368 
369   if (controlMsg.interaction_mode != NONE) {
370     this.addEventListener('mousedown', parent.startDrag.bind(parent, this));
371     this.addEventListener('mouseup', parent.stopDrag.bind(parent, this));
372     this.addEventListener('contextmenu', parent.showMenu.bind(parent, this));
373     this.addEventListener('mouseover', stopPropagation);
374     this.addEventListener('mouseout', stopPropagation);
375     this.addEventListener('click', stopPropagation);
376 
377     // hacky touch support
378     this.addEventListener('touchstart', function(event3d) {
379       console.log(event3d.domEvent);
380       if (event3d.domEvent.touches.length == 1) {
381         event3d.type = 'mousedown';
382         event3d.domEvent.button = 0;
383         that.dispatchEvent(event3d);
384       }
385     });
386     this.addEventListener('touchmove', function(event3d) {
387       if (event3d.domEvent.touches.length == 1) {
388         console.log(event3d.domEvent);
389         event3d.type = 'mousemove';
390         event3d.domEvent.button = 0;
391         that.dispatchEvent(event3d);
392       }
393     });
394     this.addEventListener('touchend', function(event3d) {
395       if (event3d.domEvent.touches.length == 0) {
396         event3d.domEvent.button = 0;
397         event3d.type = 'mouseup';
398         that.dispatchEvent(event3d);
399         event3d.type = 'click';
400         that.dispatchEvent(event3d);
401       }
402     });
403   }
404 
405   // define rotation behaviour
406   var INHERIT = 0;
407   var FIXED = 1;
408   var VIEW_FACING = 2;
409 
410   var rotInv = new THREE.Quaternion();
411   var posInv = parent.position.clone().multiplyScalar(-1);
412 
413   switch (controlMsg.orientation_mode) {
414     case INHERIT:
415       rotInv = parent.quaternion.clone().inverse();
416       that.updateMatrixWorld = function(force) {
417         ImThree.InteractiveMarkerControl.prototype.updateMatrixWorld.call(that, force);
418         that.currentControlOri.copy(that.quaternion);
419         that.currentControlOri.normalize();
420       };
421       break;
422     case FIXED:
423       that.updateMatrixWorld = function(force) {
424         that.useQuaternion = true;
425         that.quaternion = that.parent.quaternion.clone().inverse();
426         that.updateMatrix();
427         that.matrixWorldNeedsUpdate = true;
428         ImThree.InteractiveMarkerControl.prototype.updateMatrixWorld.call(that, force);
429         that.currentControlOri.copy(that.quaternion);
430       };
431       break;
432     case VIEW_FACING:
433       var independent_marker_orientation = controlMsg.independent_marker_orientation;
434       that.updateMatrixWorld = function(force) {
435 
436         camera.updateMatrixWorld();
437         var cameraRot = new THREE.Matrix4().extractRotation(camera.matrixWorld);
438 
439         var ros2Gl = new THREE.Matrix4();
440         var r90 = Math.PI * 0.5;
441         var rv = new THREE.Vector3(-r90, 0, r90);
442         ros2Gl.setRotationFromEuler(rv);
443 
444         var worldToLocal = new THREE.Matrix4();
445         worldToLocal.getInverse(that.parent.matrixWorld);
446 
447         cameraRot.multiplyMatrices(cameraRot, ros2Gl);
448         cameraRot.multiplyMatrices(worldToLocal, cameraRot);
449 
450         that.currentControlOri.setFromRotationMatrix(cameraRot);
451 
452         if (!independent_marker_orientation) {
453           that.useQuaternion = true;
454           that.quaternion.copy(that.currentControlOri);
455           that.updateMatrix();
456           that.matrixWorldNeedsUpdate = true;
457         }
458 
459         ImThree.InteractiveMarkerControl.prototype.updateMatrixWorld.call(that, force);
460       };
461       break;
462     default:
463       break;
464   }
465 
466   // create visuals (markers)
467   controlMsg.markers.forEach(function(markerMsg) {
468     var markerHelper = new ROS3D.Marker({
469       message : markerMsg,
470       path : meshBaseUrl
471     });
472 
473     if (markerMsg.header.frame_id !== "") {
474       // if the marker lives in its own coordinate frame,
475       // convert position into IM's local coordinate frame
476       markerHelper.position.add(posInv);
477       markerHelper.position.applyQuaternion(rotInv);
478       markerHelper.quaternion.multiplyQuaternions(rotInv, markerHelper.quaternion);
479       markerHelper.updateMatrixWorld();
480     }
481 
482     that.add(markerHelper);
483   });
484 
485 };

489 Menu = ImThree.Menu = function(menuEntries) {
490   var allMenus = [];
491   allMenus[0] = {
492     children : []
493   };
494 
495   THREE.EventDispatcher.call(this);
496 
497   var that = this;
498 
499   this.menuDomElem = document.createElement("div");
500   this.menuDomElem.style.position = "absolute";
501   this.menuDomElem.className = "interactive_marker_menu";
502   this.menuDomElem.addEventListener("contextmenu", function(event) {
503     event.preventDefault();
504   });
505 
506   this.overlayDomElem = document.createElement("div");
507   this.overlayDomElem.className = "interactive_marker_overlay";
508 
509   this.hideListener = this.hide.bind(this);
510   this.overlayDomElem.addEventListener("contextmenu", this.hideListener);
511   this.overlayDomElem.addEventListener("click", this.hideListener);
512 
513   // parse all entries
514   for ( var i = 0; i < menuEntries.length; i++) {
515     var entry = menuEntries[i];
516     var id = entry.id;
517     allMenus[id] = {
518       title : entry.title,
519       id : id,
520       children : []
521     };
522   }
523 
524   // link children to parents
525   for ( var i = 0; i < menuEntries.length; i++) {
526     var entry = menuEntries[i];
527     var id = entry.id;
528     var menu = allMenus[id];
529     var parent = allMenus[entry.parent_id];
530     parent.children.push(menu);
531   }
532 
533   function emitMenuSelect(menuEntry, domEvent) {
534     this.dispatchEvent({
535       type : "menu_select",
536       domEvent : domEvent,
537       id : menuEntry.id,
538       controlName : this.controlName
539     });
540     this.hide(domEvent);
541   }

543   // create html menu, starting from root (id 0)
544   function makeUl(parentDomElem, parentMenu) {
545 
546     var ulElem = document.createElement("ul");
547     parentDomElem.appendChild(ulElem);
548 
549     var children = parentMenu.children;
550 
551     for ( var i = 0; i < children.length; i++) {
552       var liElem = document.createElement("li");
553       var divElem = document.createElement("div");
554       divElem.appendChild(document.createTextNode(children[i].title));
555       ulElem.appendChild(liElem);
556       liElem.appendChild(divElem);
557 
558       if (children[i].children.length > 0) {
559         makeUl(liElem, children[i]);
560         divElem.addEventListener("click", that.hide.bind(that));
561       } else {
562         divElem.addEventListener("click", emitMenuSelect.bind(that, children[i]));
563         divElem.className = "interactive_marker_menuentry";
564       }
565     }
566 
567   }
568 
569   // construct dom element
570   makeUl(this.menuDomElem, allMenus[0]);
571 };
