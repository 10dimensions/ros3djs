1 ROS3D.InteractiveMarkerMenu = function(menuEntries) {
  2   var allMenus = [];
  3   allMenus[0] = {
  4     children : []
  5   };
  6 
  7   THREE.EventDispatcher.call(this);
  8 
  9   var that = this;
 10 
 11   this.menuDomElem = document.createElement("div");
 12   this.menuDomElem.style.position = "absolute";
 13   this.menuDomElem.className = "interactive_marker_menu";
 14   this.menuDomElem.addEventListener("contextmenu", function(event) {
 15     event.preventDefault();
 16   });
 17 
 18   this.overlayDomElem = document.createElement("div");
 19   this.overlayDomElem.className = "interactive_marker_overlay";
 20 
 21   this.hideListener = this.hide.bind(this);
 22   this.overlayDomElem.addEventListener("contextmenu", this.hideListener);
 23   this.overlayDomElem.addEventListener("click", this.hideListener);
 24 
 25   // parse all entries
 26   for ( var i = 0; i < menuEntries.length; i++) {
 27     var entry = menuEntries[i];
 28     var id = entry.id;
 29     allMenus[id] = {
 30       title : entry.title,
 31       id : id,
 32       children : []
 33     };
 34   }
 35 
 36   // link children to parents
 37   for ( var i = 0; i < menuEntries.length; i++) {
 38     var entry = menuEntries[i];
 39     var id = entry.id;
 40     var menu = allMenus[id];
 41     var parent = allMenus[entry.parent_id];
 42     parent.children.push(menu);
 43   }
 44 
 45   function emitMenuSelect(menuEntry, domEvent) {
 46     this.dispatchEvent({
 47       type : "menu_select",
 48       domEvent : domEvent,
 49       id : menuEntry.id,
 50       controlName : this.controlName
 51     });
 52     this.hide(domEvent);
 53   }
 54 
 55   // create html menu, starting from root (id 0)
 56   function makeUl(parentDomElem, parentMenu) {
 57 
 58     var ulElem = document.createElement("ul");
 59     parentDomElem.appendChild(ulElem);
 60 
 61     var children = parentMenu.children;
 62 
 63     for ( var i = 0; i < children.length; i++) {
 64       var liElem = document.createElement("li");
 65       var divElem = document.createElement("div");
 66       divElem.appendChild(document.createTextNode(children[i].title));
 67       ulElem.appendChild(liElem);
 68       liElem.appendChild(divElem);
 69 
 70       if (children[i].children.length > 0) {
 71         makeUl(liElem, children[i]);
 72         divElem.addEventListener("click", that.hide.bind(that));
 73       } else {
 74         divElem.addEventListener("click", emitMenuSelect.bind(that, children[i]));
 75         divElem.className = "interactive_marker_menuentry";
 76       }
 77     }
 78 
 79   }
 80 
 81   // construct dom element
 82   makeUl(this.menuDomElem, allMenus[0]);
 83 };
 84 
 85 ROS3D.InteractiveMarkerMenu.prototype.show = function(control, event) {
 86   if (event && event.preventDefault) {
 87     event.preventDefault();
 88   }
 89 
 90   this.controlName = control.name;
 91 
 92   // this.overlayDomElem.style.visibility = "visible";
 93   this.menuDomElem.style.left = event.domEvent.clientX + 'px';
 94   this.menuDomElem.style.top = event.domEvent.clientY + 'px';
 95   document.body.appendChild(this.overlayDomElem);
 96   document.body.appendChild(this.menuDomElem);
 97 };
 98 
 99 ROS3D.InteractiveMarkerMenu.prototype.hide = function(event) {
100   if (event && event.preventDefault) {
101     event.preventDefault();
102   }
103 
104   document.body.removeChild(this.overlayDomElem);
105   document.body.removeChild(this.menuDomElem);
106 };
