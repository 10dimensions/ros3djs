  3 ROS3D.InteractiveMarkerClient = function(options) {
  4   var that = this;
  5   this.ros = options.ros;
  6   this.tfClient = options.tfClient;
  7   this.topic = options.topic;
  8   this.root = new THREE.Object3D();
  9   this.interactiveMarkers = {};
 10   this.updateTopic = null;
 11   this.feedbackTopic = null;
 12   this.camera = options.camera;
 13   this.meshBaseUrl = '';
 14 
 15   // check for an initial topic
 16   if (this.topic) {
 17     this.subscribe(this.topic);
 18   }
 19 
 20   this.on('created_marker', function(marker) {
 21     var intMarker = new InteractiveMarker(marker, that.camera, that.meshBaseUrl);
 22     that.root.add(intMarker);
 23 
 24     marker.on('server_updated_pose', function(pose) {
 25       intMarker.onServerSetPose({
 26         pose : pose
 27       });
 28     });
 29 
 30     intMarker.addEventListener('user_changed_pose', marker.setPoseFromClient.bind(marker));
 31     intMarker.addEventListener('user_mouse_down', marker.onMouseDown.bind(marker));
 32     intMarker.addEventListener('user_mouse_up', marker.onMouseUp.bind(marker));
 33     intMarker.addEventListener('user_button_click', marker.onButtonClick.bind(marker));
 34     intMarker.addEventListener('menu_select', marker.onMenuSelect.bind(marker));
 35   });
 36 };

 39 ROS3D.InteractiveMarkerClient.prototype.subscribe = function(topic) {
 40   // unsubscribe to the other topics
 41   this.unsubscribe();
 42 
 43   this.updateTopic = new ROSLIB.Topic({
 44     ros : this.ros,
 45     name : topic + '/tunneled/update',
 46     messageType : 'visualization_msgs/InteractiveMarkerUpdate'
 47   });
 48   this.updateTopic.subscribe(this.processUpdate.bind(this));
 49 
 50   this.feedbackTopic = new ROSLIB.Topic({
 51     ros : this.ros,
 52     name : topic + '/feedback',
 53     messageType : 'visualization_msgs/InteractiveMarkerFeedback'
 54   });
 55   this.feedbackTopic.advertise();
 56 
 57   this.initService = new ROSLIB.Service({
 58     ros : this.ros,
 59     name : topic + '/tunneled/get_init',
 60     serviceType : 'demo_interactive_markers/GetInit'
 61   });
 62   var request = new ROSLIB.ServiceRequest({});
 63   this.initService.callService(request, this.processInit.bind(this));
 64 };
 65 
 66 ROS3D.InteractiveMarkerClient.prototype.unsubscribe = function() {
 67   if (this.updateTopic) {
 68     this.updateTopic.unsubscribe();
 69   }
 70   if (this.feedbackTopic) {
 71     this.feedbackTopic.unadvertise();
 72   }
 73   // erase all markers
 74   for (intMarkerName in this.interactiveMarkers) {
 75     this.eraseIntMarker(intMarkerName);
 76   }
 77   this.interactiveMarkers = {};
 78 };
 79 
 80 ROS3D.InteractiveMarkerClient.prototype.eraseIntMarker = function(intMarkerName) {
 81   if (this.interactiveMarkers[intMarkerName]) {
 82     this.emit('deleted_marker', intMarkerName);
 83     delete this.interactiveMarkers[intMarkerName];
 84   }
 85 };

 87 ROS3D.InteractiveMarkerClient.prototype.processInit = function(initMessage) {
 88   var message = initMessage.msg;
 89   var client = this;
 90   message.erases = [];
 91   for (intMarkerName in this.interactiveMarkers) {
 92     message.erases.push(intMarkerName);
 93   };
 94   message.poses = [];
 95   this.processUpdate(message);
 96 };
 97 
 98 ROS3D.InteractiveMarkerClient.prototype.processUpdate = function(message) {
 99   var that = this;
100 
101   // Deletes markers
102   message.erases.forEach(function(name) {
103     var marker = that.interactiveMarkers[name];
104     that.eraseIntMarker(name);
105   });
106 
107   // Updates marker poses
108   message.poses.forEach(function(poseMessage) {
109     var marker = that.interactiveMarkers[poseMessage.name];
110     if (marker) {
111       marker.setPoseFromServer(poseMessage.pose);
112     }
113   });
114 
115   // Adds new markers
116   message.markers.forEach(function(imMsg) {
117     var oldIntMarkerHandle = that.interactiveMarkers[imMsg.name];
118     if (oldIntMarkerHandle) {
119       that.eraseIntMarker(oldIntMarkerHandle.name);
120     }
121 
122     var intMarkerHandle = new ImProxy.IntMarkerHandle(imMsg, that.feedbackTopic, that.tfClient);
123     that.interactiveMarkers[imMsg.name] = intMarkerHandle;
124     that.emit('created_marker', intMarkerHandle);
125     // this might trigger a transform update event immediately,
126     // so we need to call it after emitting a created_marker event.
127     intMarkerHandle.subscribeTf(imMsg);
128   });
129 };

133 var IntMarkerHandle = ImProxy.IntMarkerHandle = function(imMsg, feedbackTopic, tfClient) {
134   this.feedbackTopic = feedbackTopic;
135   this.tfClient = tfClient;
136   this.name = imMsg.name;
137   this.header = imMsg.header;
138   this.controls = imMsg.controls;
139   this.menuEntries = imMsg.menu_entries;
140   this.dragging = false;
141   this.timeoutHandle = null;
142   this.tfTransform = new ROSLIB.Transform();
143   this.pose = new ROSLIB.Pose();
144   this.setPoseFromServer(imMsg.pose);
145 };

148 var KEEP_ALIVE = 0;
149 var POSE_UPDATE = 1;
150 var MENU_SELECT = 2;
151 var BUTTON_CLICK = 3;
152 var MOUSE_DOWN = 4;
153 var MOUSE_UP = 5;

155 IntMarkerHandle.prototype.subscribeTf = function(imMsg) {
156   // subscribe to tf updates if frame-fixed
157   if (imMsg.header.stamp.secs === 0.0 && imMsg.header.stamp.nsecs === 0.0) {
158     this.tfClient.subscribe(imMsg.header.frame_id, this.tfUpdate.bind(this));
159   }
160 };

162 IntMarkerHandle.prototype.emitServerPoseUpdate = function() {
163   var poseTransformed = new ROSLIB.Pose({
164     position : this.pose.position,
165     orientation : this.pose.orientation
166   });
167   poseTransformed.applyTransform(this.tfTransform);
168   this.emit('server_updated_pose', poseTransformed);
169 };
170 
171 IntMarkerHandle.prototype.setPoseFromServer = function(poseMsg) {
172   this.pose = new ROSLIB.Pose(poseMsg);
173   this.emitServerPoseUpdate();
174 };
175 
176 IntMarkerHandle.prototype.tfUpdate = function(transformMsg) {
177   this.tfTransform = new ROSLIB.Transform(transformMsg);
178   this.emitServerPoseUpdate();
179 };
180 
181 IntMarkerHandle.prototype.setPoseFromClient = function(event) {
182   this.pose = new ROSLIB.Pose(event);
183   this.pose.applyInverseTransform(this.tfTransform);
184   this.emit('client_updated_pose', event);
185   this.sendFeedback(POSE_UPDATE, undefined, 0, event.controlName);
186 
187   // keep sending pose feedback until the mouse goes up
188   if (this.dragging) {
189     if (this.timeoutHandle) {
190       clearTimeout(this.timeoutHandle);
191     }
192     this.timeoutHandle = setTimeout(this.setPoseFromClient.bind(this, event), 250);
193   }
194 };

196 IntMarkerHandle.prototype.onButtonClick = function(event) {

200 IntMarkerHandle.prototype.onMouseDown = function(event) {
201   this.sendFeedback(MOUSE_DOWN, event.clickPosition, 0, event.controlName);

205 IntMarkerHandle.prototype.onMouseUp = function(event) {
206   this.sendFeedback(MOUSE_UP, event.clickPosition, 0, event.controlName);
207   this.dragging = false;
208   if (this.timeoutHandle) {
209     clearTimeout(this.timeoutHandle);
210   }
211 };

213 IntMarkerHandle.prototype.onMenuSelect = function(event) {

217 IntMarkerHandle.prototype.sendFeedback = function(eventType, clickPosition, menu_entry_id,
218     controlName) {
219 
220   var mouse_point_valid = clickPosition !== undefined;
221   var clickPosition = clickPosition || {
222     x : 0,
223     y : 0,
224     z : 0
225   };
226 
227   var feedback = {
228     header : this.header,
229     client_id : this.client_id,
230     marker_name : this.name,
231     control_name : controlName,
232     event_type : eventType,
233     pose : this.pose,
234     mouse_point : clickPosition,
235     mouse_point_valid : mouse_point_valid,
236     menu_entry_id : menu_entry_id
237   };
238   this.feedbackTopic.publish(feedback);
239 };
